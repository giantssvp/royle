CREATE TABLE common_db.royal_contact ( ID int NOT NULL AUTO_INCREMENT, Name varchar(255), Email_id varchar(255), Phone varchar(255), Subject varchar(255), Message varchar(2048) NOT NULL, Date date NOT NULL, Status boolean NOT NULL, PRIMARY KEY (ID));CREATE TABLE common_db.royal_transport ( ID int NOT NULL AUTO_INCREMENT, Name varchar(255), Email_id varchar(255), Phone varchar(255), transport_from varchar(255), transport_to varchar(255), Date date NOT NULL, Status boolean NOT NULL, PRIMARY KEY (ID));
